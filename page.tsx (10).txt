"use client"

import { useState, useEffect } from "react"
import Link from "next/link"
import { Button } from "@/components/ui/button"
import { Card, CardContent, CardFooter, CardHeader, CardTitle } from "@/components/ui/card"
import { Input } from "@/components/ui/input"
import { Label } from "@/components/ui/label"
import { Plane, Calendar, ArrowRight, Search } from "lucide-react"
import { getAllFlights } from "@/lib/data"

export default function FlightsPage() {
  const [flights, setFlights] = useState([])
  const [filteredFlights, setFilteredFlights] = useState([])
  const [searchParams, setSearchParams] = useState({
    from: "",
    to: "",
    date: "",
  })
  const [loading, setLoading] = useState(true)

  useEffect(() => {
    const fetchFlights = async () => {
      const data = await getAllFlights()
      setFlights(data)
      setFilteredFlights(data)
      setLoading(false)
    }

    fetchFlights()
  }, [])

  const handleInputChange = (e) => {
    const { id, value } = e.target
    setSearchParams((prev) => ({
      ...prev,
      [id]: value,
    }))
  }

  const handleSearch = () => {
    let results = [...flights]

    if (searchParams.from) {
      results = results.filter((flight) => flight.departureCity.toLowerCase().includes(searchParams.from.toLowerCase()))
    }

    if (searchParams.to) {
      results = results.filter((flight) => flight.arrivalCity.toLowerCase().includes(searchParams.to.toLowerCase()))
    }

    if (searchParams.date) {
      results = results.filter((flight) => flight.date === searchParams.date)
    }

    setFilteredFlights(results)
  }

  return (
    <div className="container py-10">
      <h1 className="text-3xl font-bold mb-6 text-primary">Available Flights</h1>

      <div className="mb-8 p-6 bg-[#dbeafe] rounded-lg shadow-md">
        <h2 className="text-xl font-semibold mb-4 text-primary">Search Flights</h2>
        <div className="grid gap-4 md:grid-cols-4">
          <div className="space-y-2">
            <Label htmlFor="from" className="text-gray-700">
              From
            </Label>
            <Input
              id="from"
              placeholder="Departure city"
              className="border-blue-200 focus:border-blue-500"
              value={searchParams.from}
              onChange={handleInputChange}
            />
          </div>
          <div className="space-y-2">
            <Label htmlFor="to" className="text-gray-700">
              To
            </Label>
            <Input
              id="to"
              placeholder="Arrival city"
              className="border-blue-200 focus:border-blue-500"
              value={searchParams.to}
              onChange={handleInputChange}
            />
          </div>
          <div className="space-y-2">
            <Label htmlFor="date" className="text-gray-700">
              Date
            </Label>
            <Input
              id="date"
              type="date"
              className="border-blue-200 focus:border-blue-500"
              value={searchParams.date}
              onChange={handleInputChange}
            />
          </div>
          <div className="flex items-end">
            <Button className="w-full bg-primary hover:bg-secondary transition-colors" onClick={handleSearch}>
              <Search className="h-4 w-4 mr-2" />
              Search
            </Button>
          </div>
        </div>
      </div>

      {loading ? (
        <div className="flex justify-center items-center h-40">
          <div className="animate-spin rounded-full h-12 w-12 border-t-2 border-b-2 border-primary"></div>
        </div>
      ) : filteredFlights.length === 0 ? (
        <div className="text-center py-10">
          <h3 className="text-xl font-medium text-gray-600">No flights found matching your criteria</h3>
          <p className="text-gray-500 mt-2">Try adjusting your search parameters</p>
        </div>
      ) : (
        <div className="grid gap-6 md:grid-cols-2 lg:grid-cols-3">
          {filteredFlights.map((flight) => (
            <Card key={flight.id} className="border-2 border-blue-100 shadow-md hover:shadow-lg transition-all">
              <CardHeader>
                <CardTitle className="flex justify-between items-center">
                  <div className="flex items-center gap-2 text-primary">
                    <Plane className="h-5 w-5 text-primary" />
                    {flight.flightNumber}
                  </div>
                  <span className="text-lg font-bold text-primary">${flight.price}</span>
                </CardTitle>
              </CardHeader>
              <CardContent>
                <div className="flex justify-between items-center mb-4">
                  <div>
                    <p className="text-2xl font-bold text-gray-800">{flight.departureCity}</p>
                    <p className="text-sm text-gray-600">{flight.departureTime}</p>
                  </div>
                  <ArrowRight className="h-5 w-5 mx-2 text-blue-500" />
                  <div>
                    <p className="text-2xl font-bold text-gray-800">{flight.arrivalCity}</p>
                    <p className="text-sm text-gray-600">{flight.arrivalTime}</p>
                  </div>
                </div>
                <div className="flex items-center gap-2 text-sm text-gray-600">
                  <Calendar className="h-4 w-4 text-blue-500" />
                  <span>{flight.date}</span>
                </div>
                <div className="mt-2 text-sm text-gray-600">
                  <span className="font-medium">{flight.availableSeats}</span> seats available
                </div>
              </CardContent>
              <CardFooter>
                <Button asChild className="w-full bg-primary hover:bg-secondary transition-colors">
                  <Link href={`/bookings/new?flightId=${flight.id}`}>Book Now</Link>
                </Button>
              </CardFooter>
            </Card>
          ))}
        </div>
      )}
    </div>
  )
}
